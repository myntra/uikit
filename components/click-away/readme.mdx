const ref = useRef();
const [message, setMessage] = useState('Initial State.');
const handleInside = useCallback(() => setMessage('Clicked Inside.'), [setMessage]);
const handleOutside = useCallback(() => setMessage('Clicked Outside.'), [setMessage]);

<div>
  <p>Status: {message}</p>

  <div
    ref={ref}
    onClick={handleInside}
    style={{ padding: '20px', background: '#eee', width: 250 }}
  >

  </div>

  <ClickAway onClickAway={handleOutside} target={ref} />
</div>
